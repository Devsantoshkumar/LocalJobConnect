<script>

const textContainer = document.getElementById('text-container');
const homeImage = document.getElementById('home-image');
let index = 0; // Current text index
let intervalId = null; // Variable to store the interval ID

// Define an array of text and corresponding image URLs
const textData = [
  { text: 'India', image: '<?=ASSETS ?>images/india.png' },
  { text: 'Singapore', image: '<?=ASSETS ?>images/image-singapore.jpg' },
  { text: 'Maldives', image: '<?=ASSETS ?>images/image-maldives.jpg' },
  { text: 'Dubai', image: '<?=ASSETS ?>images/image-dubai.jpg' },
  { text: 'Sri Lanka', image: '<?=ASSETS ?>images/image-shrilanka.jpg' },
  { text: 'Thailand', image: '<?=ASSETS ?>images/image-thailand.jpg' },
  { text: 'Bhutan', image: '<?=ASSETS ?>images/image-bhutan.jpg' }
];

// Function to update the text with typing effect
function updateTextWithTypingEffect() {
  const { text, image } = textData[index];
  let i = 0;
  textContainer.textContent = '';

  // Change the image before starting the typing effect
  homeImage.src = image;

  function type() {
    if (i < text.length) {
      textContainer.textContent += text.charAt(i);
      i++;
      setTimeout(type, 200); // Adjust typing speed (milliseconds)
    }
  }

  type();

  // Increment the index to move to the next text
  index = (index + 1) % textData.length;
}

// Function to start the interval for text change
function startInterval() {
  intervalId = setInterval(updateTextWithTypingEffect, 3500); // Adjust text change interval (milliseconds)
}

// Function to stop the interval
function stopInterval() {
  clearInterval(intervalId);
}

// Call the function initially
updateTextWithTypingEffect();

// Call the function periodically to change the text
startInterval();

// Pause/resume the animation when the page visibility changes
document.addEventListener('visibilitychange', function () {
  if (document.visibilityState === 'visible') {
    startInterval();
  } else {
    stopInterval();
  }
});


</script>
